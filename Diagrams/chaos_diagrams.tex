% Double pendulum
\tikzmath{\a1 = 60; \a2 = 40; \l1 = 3; \l2 = 3;}

\begin{tikzpicture}
    \coordinate (0) at (0, 0);
    \coordinate (1) at ($ (0)+({\a1 + 270}:\l1) $);
    \coordinate (2) at ($ (1)+({\a2 + 270}:\l2) $);
    \draw [dashed] (0) -- ++(270:{\l1 + 0.5});
    \draw [blue,line width=2pt] (0) -- (1)  node [black,pos=0.5,above right] {$L_1$};
    \draw [red,line width=2pt] (1) -- (2) node [black,pos=0.5,above right] {$L_2$};
    \draw [fill] (0) circle (5pt);
    \node [white,fill=blue,circle,inner sep=2pt] at (1) {$\boldsymbol{m_1}$};
    \node [white,fill=red,circle,inner sep=2pt] at (2) {$\boldsymbol{m_2}$};
    \node at ($ (0)+({270 + 0.5*\a1}:0.9) $) {$\theta_1$};
    \node at ($ (1)+({270 + 0.5*\a2}:0.9) $) {$\theta_2$};
    \draw [-triangle 45,shorten <= 10pt] (1) -- node [left] {$g$} ++(270:1.75);
    \draw [-triangle 45,shorten <= 10pt] (2) -- node [left] {$g$} ++(270:1.75);
    \draw ($ (0)+({270 + \a1}:1.25) $) arc ({270 + \a1}:270:1.25);
    \draw ($ (1)+({270 + \a2}:1.25) $) arc ({270 + \a2}:270:1.25);
\end{tikzpicture}
